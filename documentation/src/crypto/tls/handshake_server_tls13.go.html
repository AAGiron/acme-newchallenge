func ( *serverHandshakeStateTLS13) () error {
	 := .c

	// For an overview of the TLS 1.3 handshake, see RFC 8446, Section 2.
	if  := .processClientHello();  != nil {
		return 
	}
	if  := .checkForResumption();  != nil {
		return 
	}
	if  := .pickCertificate();  != nil {
		return 
	}
	.buffering = true
	if  := .sendServerParameters();  != nil {
		return 
	}
	if  := .sendServerCertificate();  != nil {
		return 
	}
	if .isKEMTLS {
		// send application data for KEMTLS
		if !.pdkKEMTLS {
			if ,  := .flush();  != nil {
				return 
			}
			// second round of sending for KEMTLS
			.handshakeTimings.reset()
		}

		return .handshakeKEMTLS()
	}
	if  := .sendServerFinished();  != nil {
		return 
	}
	if !.pdkKEMTLS {
		// Note that at this point we could start sending application data without
		// waiting for the client's second flight, but the application might not
		// expect the lack of replay protection of the ClientHello parameters.
		if ,  := .flush();  != nil {
			return 
		}

		// second round of TLS1.3, thrid round for KEMTLS and fourth round for mKEMTLS
		.handshakeTimings.reset()
	}

	if  := .readClientCertificate();  != nil {
		return 
	}
	if  := .readClientFinished();  != nil {
		return 
	}

	//hs.handshakeTimings.ExperimentName = experimentName(c)
	// fmt.Printf("\n %+v \n", hs.handshakeTimings)
	.handshakeTimings.finish()
	.handleCFEvent(.handshakeTimings)

	atomic.StoreUint32(&.handshakeStatus, 1)

	return nil
}

// interfaces implemented by standard library hashes to clone the state of in

	}

	}

	}

// getDelegatedCredential will return a Delegated Credential pair (a Delegated
// Credential and its private key) for the given ClientHelloInfo, defaulting to
// the first element of cert.DelegatedCredentialPair.
// The returned Delegated Credential could be invalid for usage in the handshake.
// Returns an error if there are no delegated credentials or if the one found
// cannot be used for the current connection.
func ( *ClientHelloInfo,  *Certificate) (*DelegatedCredentialPair, error) {
	if len(.DelegatedCredentials) == 0 {
		return nil, errors.New("No Delegated Credential found.")
	}

	if len(.DelegatedCredentials) == 1 {
		// There's only one choice, so no point doing any work.
		return &.DelegatedCredentials[0], nil
	}

	for ,  := range .DelegatedCredentials {
		// The client must have sent the signature_algorithms in the DC extension: ensure it supports
		// schemes we can use with this delegated credential.
		if len(.SignatureSchemesDC) > 0 {
			if ,  := selectSignatureSchemeDC(VersionTLS13, .DC, .SignatureSchemesDC);  == nil {
				return &, nil
			}
		}
	}

	// No delegated credential can be returned.
	return nil, errors.New("No valid Delegated Credential found.")
}

// pickCertificate selects the server's certificate to be used in the TLS handshake..
// Additionally, the server will staple the dummy OCSP response generated by the ACME server.
func ( *serverHandshakeStateTLS13) () error {
	 := .c

	// Only one of PSK and certificates are used at a time.
	if .usingPSK {
		return nil
	}

	if .pdkKEMTLS && .keyKEMShare {
		return nil
	}

	if .usingCachedInformation {
		.didPQTLS = true
		return nil
	}

	// signature_algorithms is required in TLS 1.3. See RFC 8446, Section 4.2.3.
	if len(.clientHello.supportedSignatureAlgorithms) == 0 {
		return .sendAlert(alertMissingExtension)
	}

	var  *Certificate	
	var  error

	,  = .config.getCertificate(clientHelloInfo(, .clientHello))
	if  != nil {
		if  == errNoCertificates {
			.sendAlert(alertUnrecognizedName)
		} else {
			.sendAlert(alertInternalError)
		}
		return 
	}
	

	.sigAlg,  = selectSignatureScheme(.vers, , .clientHello.supportedSignatureAlgorithms)
	if  != nil {
		// getCertificate returned a certificate that is unsupported or
		// incompatible with the client's signature algorithms.
		.sendAlert(alertHandshakeFailure)
		return 
	}

	.cert = 

	if .clientHello.delegatedCredentialSupported && len(.clientHello.supportedSignatureAlgorithmsDC) > 0 {
		,  := getDelegatedCredential(clientHelloInfo(, .clientHello), .cert)
		if  != nil {
			// a Delegated Credential was not found. Fallback to the certificate.
			return nil
		}

		if .DC != nil && .PrivateKey != nil {
			// Even if the Delegated Credential has already been marshalled, be sure it is the correct one.
			.DC.raw,  = .DC.marshal()
			if  != nil {
				// invalid Delegated Credential. Fallback to the certificate.
				return nil
			}

			.sigAlg,  = selectSignatureSchemeDC(.vers, .DC, .clientHello.supportedSignatureAlgorithmsDC)
			if  != nil {
				// the Delegated Credential is unsupported or
				// incompatible with the client's signature algorithms.
				// Fallback to the certificate.
				return nil
			}

			.cert.DelegatedCredentialPrivateKey = .PrivateKey
			.cert.DelegatedCredential = .DC.raw
			if .keyKEMShare {
				if .DC.cred.expCertVerfAlgo.isKEMTLS() {
					.isKEMTLS = true
				} else if .DC.cred.expCertVerfAlgo.isPQTLS() {
					.didPQTLS = true
				} else {
					return nil
				}
			}
		}
	} else {
		if .keyKEMShare {
			if .sigAlg.isKEMTLS() {
				.isKEMTLS = true
			} else if .sigAlg.isPQTLS() {
				.didPQTLS = true
			// TODO:VVC: Understand why is not going to the above case
			} else if .c.config.PQTLSEnabled {
				.didPQTLS = true
			} else {
				.sendAlert(alertHandshakeFailure)
				return 
			}
		}
	}

	return nil
}
